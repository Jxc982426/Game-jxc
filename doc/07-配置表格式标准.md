# 07-配置表格式标准

## 📋 文档概述

本文档定义了游戏开发中使用的所有配置表的格式标准，包括 JSON 数据结构、字段命名规范、ID 编码规则等。所有开发人员必须遵循此标准以确保数据一致性。

**创建日期：** 2026-04-02  
**适用版本：** v1.0

---

## 一、通用命名规范

### 1.1 字段命名规则

```
命名风格：snake_case（小写 + 下划线）
示例：item_id, max_health, damage_multiplier

布尔类型：is_, has_, can_ 前缀
示例：is_active, has_effect, can_stack

数值类型：明确单位
示例：damage_value, health_percent, cooldown_seconds

数组类型：复数形式
示例：skills, affixes, drop_items
```

### 1.2 ID 编码规范

```
格式：[类别前缀]_[子类别]_[序号]

物品 ID: item_[type]_[id]
- item_wep_001 (武器 - 剑 -001)
- item_arm_002 (防具 - 衣服 -002)
- item_pot_003 (消耗品 - 药水 -003)

敌人 ID: enemy_[zone]_[id]
- enemy_town_001 (城镇区域 -001)
- enemy_dun1_002 (地下城 1-002)
- enemy_boss_001 (BOSS-001)

技能 ID: skill_[class]_[id]
- skill_war_001 (战士技能 -001)
- skill_mag_002 (法师技能 -002)
- skill_nec_003 (死灵技能 -003)

任务 ID: quest_[type]_[id]
- quest_main_001 (主线任务 -001)
- quest_side_002 (支线任务 -002)
- quest_daily_003 (每日任务 -003)

地图 ID: map_[zone]_[id]
- map_town_001 (城镇 -001)
- map_dun1_001 (地下城 1-001)
```

import json
import jsonschema

def validate_items_config(data):
    schema = {
        "type": "object",
        "properties": {
            "items": {
                "type": "array",
                "items": {
                    "type": "object",
                    "required": ["item_id", "item_name", "item_type", "quality"],

                        "item_id": {"type": "string", "pattern": "^item_[a-z]+_[0-9]+$"},
                        "quality": {"type": "integer", "minimum": 0, "maximum": 3},
                        "max_stack": {"type": "integer", "minimum": 1}
                    }
                }
            }
        }
    }
    jsonschema.validate(data, schema)

# 批量验证所有配置
def validate_all_configs():
    configs = [
        ("items.json", validate_items_config),
        ("enemies.json", validate_enemies_config),
        ("skills.json", validate_skills_config)
    ]
    
    for config_file, validator in configs:

            data = json.load(f)
            validator(data)
            print(f"✅ {config_file} 验证通过")

## 九、配置表版本管理

### 9.1 版本号规范

```
版本格式：v{主版本}.{次版本}.{修订号}
示例：v1.0.0, v1.2.5, v2.0.0

版本规则:
- 主版本号：不兼容的格式变更
- 次版本号：向后兼容的功能新增
- 修订号：向后兼容的问题修复

示例:
v1.0.0 → v1.0.1  # 修复配置错误
v1.0.1 → v1.1.0  # 新增词缀类型
v1.1.0 → v2.0.0  # 配置结构重构
```

    var file = FileAccess.open(config_path, FileAccess.READ)

    print("开始配置迁移：%s → %s" % [from_version, current_config_version])

        config = migration_func.call(config)

    return config

# 示例：v1.0.0 到 v2.0.0 的迁移
func migrate_v1_to_v2(config: Dictionary) -> Dictionary:
    # 添加新的品质等级（传奇）
    for item in config.items:
        if not item.has("rarity_power"):
            item["rarity_power"] = 0
    
    # 更新元数据
    config["_meta"]["version"] = "v2.0.0"
    config["_meta"]["updated_at"] = Time.get_datetime_string_from_system()
    
    return config

### 9.4 配置差异对比工具

``python
# diff_configs.py - 配置差异对比工具
import json
import sys

def compare_configs(old_path: str, new_path: str):
    with open(old_path, 'r', encoding='utf-8') as f:
        old_data = json.load(f)
    
    with open(new_path, 'r', encoding='utf-8') as f:
        new_data = json.load(f)
    
    # 比较元数据
    old_version = old_data.get('_meta', {}).get('version', 'unknown')
    new_version = new_data.get('_meta', {}).get('version', 'unknown')
    
    print(f"版本对比：{old_version} → {new_version}")
    print("=" * 50)
    
    # 比较物品数量
    old_count = len(old_data.get('items', []))
    new_count = len(new_data.get('items', []))
    print(f"物品数量：{old_count} → {new_count} ({'+' if new_count > old_count else ''}{new_count - old_count})")
    
    # 找出新增的物品
    old_ids = {item['item_id'] for item in old_data.get('items', [])}
    new_ids = {item['item_id'] for item in new_data.get('items', [])}
    
    added = new_ids - old_ids
    removed = old_ids - new_ids
    modified = []
    
    # 检查修改的物品
    for new_item in new_data.get('items', []):
        if new_item['item_id'] in old_ids:
            old_item = next((i for i in old_data['items'] if i['item_id'] == new_item['item_id']), None)
            if old_item and old_item != new_item:
                modified.append(new_item['item_id'])
    
    print(f"\n新增物品：{len(added)}")
    for item_id in sorted(added):
        print(f"  + {item_id}")
    
    print(f"\n移除物品：{len(removed)}")
    for item_id in sorted(removed):
        print(f"  - {item_id}")
    
    print(f"\n修改物品：{len(modified)}")
    for item_id in sorted(modified):
        print(f"  ~ {item_id}")

if __name__ == '__main__':
    if len(sys.argv) != 3:
        print("用法：python diff_configs.py <旧文件> <新文件>")
        sys.exit(1)
    
    compare_configs(sys.argv[1], sys.argv[2])
```

### 10.1 运行时重新加载

``gdscript
# HotReloadManager.gd - 配置热重载管理器
extends Node

signal config_reloaded(config_name: String, success: bool)

var config_cache: Dictionary = {}
var file_watchers: Dictionary = {}
var auto_reload_enabled = true

func _ready():
    # 在编辑器中启用自动重载
    if Engine.is_editor_hint():
        enable_auto_reload()

func enable_auto_reload():
    # 监控配置文件夹
    var config_dir = "res://config/"
    var dir = DirAccess.open(config_dir)
    
    if dir:
        dir.list_dir_begin()
        var file_name = dir.get_next()
        
        while file_name != "":
            if file_name.ends_with(".json"):
                watch_file(config_dir + file_name)
            file_name = dir.get_next()

class ConfigWatcher:
    var file_path: String
    var last_modified_time: int
    var reload_callback: Callable
    
    func _init(path: String, callback: Callable):
        file_path = path
        reload_callback = callback
        update_last_modified()
    
    func check_for_changes() -> bool:
        var current_time = get_file_modified_time(file_path)
        if current_time != last_modified_time:
            last_modified_time = current_time
            return true
        return false
    
    func update_last_modified():
        last_modified_time = get_file_modified_time(file_path)

var watchers: Array[ConfigWatcher] = []

func watch_file(file_path: String):
    var watcher = ConfigWatcher.new(file_path, _on_config_changed)
    watchers.append(watcher)

func _process(_delta):
    if not auto_reload_enabled:
        return
    
    # 每 2 秒检查一次文件变化
    if Engine.get_frames_drawn() % 120 == 0:
        for watcher in watchers:
            if watcher.check_for_changes():
                print("检测到配置变更：", watcher.file_path)
                watcher.reload_callback.call(watcher.file_path)

func _on_config_changed(file_path: String):
    var config_name = file_path.get_file().replace(".json", "")
    
    # 重新加载配置
    var new_config = load_config(file_path)
    
    if new_config:
        config_cache[config_name] = new_config
        config_reloaded.emit(config_name, true)
        print("✓ 配置已热重载：", config_name)
    else:
        config_reloaded.emit(config_name, false)
        print_error("✗ 配置热重载失败：", config_name)

# 获取配置（带缓存）
func get_config(config_name: String) -> Dictionary:
    if config_cache.has(config_name):
        return config_cache[config_name]
    
    # 从文件加载
    var file_path = "res://config/%s.json" % config_name
    if FileAccess.file_exists(file_path):
        var config = load_config(file_path)
        config_cache[config_name] = config
        return config
    
    push_error("配置不存在：", config_name)
    return {}

# 手动触发重载
func reload_config(config_name: String) -> bool:
    var file_path = "res://config/%s.json" % config_name
    if FileAccess.file_exists(file_path):
        _on_config_changed(file_path)
        return true
    return false

### 10.3 配置回滚机制

``gdscript
# ConfigRollback.gd - 配置回滚工具
extends Node

var backup_directory = "user://config_backups/"
var max_backups = 10

# 创建备份
func create_backup(config_name: String) -> String:
    var source_path = "res://config/%s.json" % config_name
    var timestamp = Time.get_datetime_string_from_system(false, true).replace(":", "-")
    var backup_path = backup_directory + "%s_%s.json" % [config_name, timestamp]
    
    # 确保备份目录存在
    DirAccess.make_dir_recursive_absolute(backup_directory)
    
    # 复制文件
    if DirAccess.copy(source_path, backup_path) == OK:
        print("备份创建成功：", backup_path)
        cleanup_old_backups(config_name)
        return backup_path
    else:
        print_error("备份创建失败：", source_path)
        return ""

# 回滚到指定版本
func rollback_to(config_name: String, backup_path: String) -> bool:
    var target_path = "res://config/%s.json" % config_name
    
    if not FileAccess.file_exists(backup_path):
        print_error("备份文件不存在：", backup_path)
        return false
    
    # 先备份当前版本
    create_backup(config_name)
    
    # 恢复备份
    if DirAccess.copy(backup_path, target_path) == OK:
        print("回滚成功：%s → %s" % [target_path, backup_path])
        
        # 触发热重载
        HotReloadManager.reload_config(config_name)
        return true
    else:
        print_error("回滚失败")
        return false

# 列出所有可用的备份
func list_backups(config_name: String) -> Array:
    var backups = []
    var dir = DirAccess.open(backup_directory)
    
    if dir:
        dir.list_dir_begin()
        var file_name = dir.get_next()
        
        while file_name != "":
            if file_name.begins_with(config_name + "_"):
                backups.append({
                    "file": file_name,
                    "path": backup_directory + file_name,
                    "time": extract_timestamp(file_name)
                })
            file_name = dir.get_next()
    
    # 按时间倒序排序
    backups.sort_custom(func(a, b): return a.time > b.time)
    return backups

func cleanup_old_backups(config_name: String):
    var backups = list_backups(config_name)
    
    # 保留最新的 N 个备份
    while backups.size() > max_backups:
        var oldest = backups.pop_back()
        DirAccess.remove_absolute(oldest.path)
        print("清理旧备份：", oldest.path)
```
