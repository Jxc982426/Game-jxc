## 二、音频播放规则

### 2.1 BGM 播放规则

```gdscript
# BGM 管理器核心逻辑
class_name BGMManager
extends Node

# BGM 淡入淡出时间
const FADE_TIME = 1.0  # 秒

# 当前播放的 BGM
var current_bgm: String = ""
var bgm_players: Dictionary = {}

# 播放 BGM（支持淡入淡出）
func play_bgm(bgm_id: String, volume_db: float = 0.0):
    if current_bgm == bgm_id:
        return
    
    # 停止当前 BGM（淡出）
    if current_bgm != "":
        fade_out(current_bgm)
    
    # 加载新 BGM（淡入）
    var audio_player = get_audio_player(bgm_id)
    audio_player.volume_db = -40.0  # 静音开始
    audio_player.play()
    
    fade_in(bgm_id, volume_db)
    current_bgm = bgm_id

# 淡入效果
func fade_in(bgm_id: String, target_volume: float):
    var tween = create_tween()
    var player = bgm_players[bgm_id]
    tween.tween_property(player, "volume_db", target_volume, FADE_TIME)

# 淡出效果
func fade_out(bgm_id: String):
    var tween = create_tween()
    var player = bgm_players[bgm_id]
    tween.tween_property(player, "volume_db", -40.0, FADE_TIME)
    tween.tween_callback(player.stop)
```

### 2.2 SFX 播放规则

#### 优先级系统

```gdscript
# SFX 优先级定义
enum SFX_PRIORITY {
    LOW = 0,      # 环境音、脚步声
    NORMAL = 1,   # 普通攻击、UI 音效
    HIGH = 2,     # 技能音效、重要提示
    CRITICAL = 3  # BOSS 咆哮、警告音
}

# 并发限制
var max_concurrent_sfx = {
    SFX_PRIORITY.LOW: 5,
    SFX_PRIORITY.NORMAL: 10,
    SFX_PRIORITY.HIGH: 15,
    SFX_PRIORITY.CRITICAL: 999  # 无限制
}

# 播放 SFX（带优先级和并发控制）
func play_sfx(sfx_id: String, priority: int = SFX_PRIORITY.NORMAL, 
              volume_db: float = 0.0, pitch_variance: float = 0.0):
    
    # 检查并发限制
    if get_playing_count(priority) >= max_concurrent_sfx[priority]:
        # 停止最早的低优先级音效
        stop_oldest_sfx(priority)
    
    # 获取音频播放器
    var player = get_available_player()
    if not player:
        return
    
    # 加载并播放
    player.stream = load_audio_stream(sfx_id)
    player.volume_db = volume_db
    
    # 随机音调（增加变化）
    if pitch_variance > 0:
        player.pitch_scale = 1.0 + randf_range(-pitch_variance, pitch_variance)
    
    player.play()
```

## 三、音频资源技术规范

### 3.1 音频格式要求

| 类型 | 格式 | 采样率 | 位深度 | 压缩质量 |
|------|------|--------|--------|----------|
| BGM | OGG Vorbis | 44.1kHz | 16bit | 80% (q=8) |
| SFX | WAV | 44.1kHz | 16bit | 无损 |
| 语音 | OGG Vorbis | 44.1kHz | 16bit | 90% (q=9) |
| 环境音 | OGG Vorbis | 44.1kHz | 16bit | 70% (q=7) |

### 3.2 文件大小限制

| 类型 | 单文件最大 | 建议大小 | 总容量限制 |
|------|-----------|----------|-----------|
| BGM | 10MB | 3-5MB | 100MB |
| SFX | 500KB | 50-200KB | 50MB |
| 语音 | 2MB | 500KB | 200MB |

### 3.3 命名规范

```
格式：[类型]_[分类]_[描述]_[变体].wav/ogg

示例:
BGM_Town_Peaceful_v1.ogg

SFX_Skill_Fireball_Cast.wav
```

## 五、音频性能优化

### 5.1 对象池管理

```gdscript
# 音频播放器对象池
var audio_pool_size = 20
var audio_players: Array[AudioStreamPlayer] = []
var available_players: Array[AudioStreamPlayer] = []

func _ready():
    for i in range(audio_pool_size):
        var player = AudioStreamPlayer.new()
        add_child(player)
        audio_players.append(player)
        available_players.append(player)

func get_available_player() -> AudioStreamPlayer:
    if available_players.is_empty():
        # 回收最早播放的
        recycle_oldest_player()
    return available_players.pop_front()

func recycle_player(player: AudioStreamPlayer):
    player.stop()
    available_players.append(player)
```

### 5.2 预加载策略

```gdscript
# 关键音效预加载
```

## 六、辅助功能

### 6.1 听力保护

```gdscript
# 最大音量限制
const MAX_VOLUME_DB = -3.0

# 突然大音量抑制
func clamp_sudden_volume(new_volume: float) -> float:
    return min(new_volume, MAX_VOLUME_DB)

# 长时间高音量警告
var high_volume_duration = 0.0
func monitor_volume_exposure():
    if get_master_volume() > -10.0:
        high_volume_duration += get_process_delta_time()
        if high_volume_duration > 3600:  # 1 小时
            show_hearing_protection_warning()
```
